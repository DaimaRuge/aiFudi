   ```python
   # services/ner/entity_extractor.py
   class EntityExtractor:
       def __init__(self):
           self.spacy_model = spacy.load("zh_core_web_lg")
           self.stanza_model = stanza.Pipeline('zh')
       
       def extract(self, text: str) -> List[Entity]:
           # 使用多种NLP工具提取实体
           entities = []
           
           # spaCy提取
           doc = self.spacy_model(text)
           for ent in doc.ents:
               entities.append(Entity(
                   text=ent.text,
                   type=ent.label_,
                   start=ent.start_char,
                   end=ent.end_char
               ))
           
           return entities
   ```

   ```python
   # services/search/vector_search.py
   class VectorSearch:
       def __init__(self):
           self.embedding_model = OpenAIEmbedding()
           self.vector_db = ChromaDB()
       
       def search(self, query: str, k: int = 10):
           # 生成查询向量
           query_embedding = self.embedding_model.embed(query)
           
           # 向量检索
           results = self.vector_db.search(
               query_embedding=query_embedding,
               k=k
           )
           
           return results
   ```

3. **混合搜索算法** (第12周)
   ```python
   # services/search/hybrid_search.py
   class HybridSearch:
       def search(self, query: str, weights: Dict[str, float] = None):
           # 并行执行多种搜索
           with ThreadPoolExecutor() as executor:
               keyword_future = executor.submit(self.keyword_search, query)
               vector_future = executor.submit(self.vector_search, query)
               graph_future = executor.submit(self.graph_search, query)
               
               keyword_results = keyword_future.result()
               vector_results = vector_future.result()
               graph_results = graph_future.result()
           
           # 结果融合
           fused_results = self.fuse_results(
               keyword_results,
               vector_results,
               graph_results,
               weights
           )
           
           return fused_results
   ```

4. **搜索API设计** (第12周)
   ```python
   # api/search.py
   @router.get("/search")
   async def search_documents(
       query: str,
       search_type: SearchType = SearchType.HYBRID,
       limit: int = 20,
       current_user: User = Depends(get_current_user)
   ):
       if search_type == SearchType.KEYWORD:
           results = keyword_search(query, limit)
       elif search_type == SearchType.VECTOR:
           results = vector_search(query, limit)
       else:  # HYBRID
           results = hybrid_search(query, limit)
       
       return {
           "query": query,
           "results": results,
           "total": len(results)
       }
   ```
